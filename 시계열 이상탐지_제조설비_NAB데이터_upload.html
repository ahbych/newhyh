In [39]:
WINDOW = 50
STRIDE = 1
def make_windows(x, window=50, stride=1):
    X = []
    for i in range(0, len(x) - window + 1, stride):
        X.append(x[i:i+window])
    return np.stack(X)
X_train = make_windows(train_scaled, WINDOW, STRIDE)  
X_test  = make_windows(test_scaled,  WINDOW, STRIDE)

n_features = 1
inp = layers.Input(shape=(WINDOW, n_features))  
x = layers.LSTM(64, return_sequences=True)(inp) 
x = layers.LSTM(32, return_sequences=False)(x)  
x = layers.RepeatVector(WINDOW)(x)
x = layers.LSTM(32, return_sequences=True)(x) # (50, 32)
x = layers.LSTM(64, return_sequences=True)(x) # (50, 64)
out = layers.TimeDistributed(layers.Dense(1))(x) # (50, 1)

ae = Model(inp, out)
ae.compile(optimizer="adam", loss="mse")

7) 재구성오차 + threshold + 이상판정

In [44]:
train_rec = ae.predict(X_train, verbose=0)
train_err = np.mean((train_rec - X_train) ** 2, axis=(1, 2))# 각 윈도우의 각 시점에서 오차의 제곱 평균
threshold = np.quantile(train_err, 0.995)                    # 9.5%보다 큰 오차면 정상으로 보기 어렵다
print("threshold:", threshold)
# 테스트 데이터
test_rec = ae.predict(X_test, verbose=0)
test_err = np.mean((test_rec - X_test) ** 2, axis=(1, 2))

anomaly_flag = (test_err > threshold).astype(int) 
print("anomaly windows:", anomaly_flag.sum(), "/", len(anomaly_flag))

# anomaly_flag(=result_df["anomaly"])에서 연속 구간(이벤트) 뽑기
a = result_df["anomaly"].to_numpy()
intervals = []
in_run = False
start = None

for i, v in enumerate(a):
    if v == 1 and not in_run:
        in_run = True
        start = i
    if v == 0 and in_run:
        in_run = False
        intervals.append((start, i-1))

if in_run:
    intervals.append((start, len(a)-1))

# 재구성오차 그래프에 이벤트 구간 음영 처리
plt.figure(figsize=(14, 4))
plt.plot(result_df["timestamp"], result_df["recon_error"])
plt.axhline(threshold, linestyle="--")

for s, e in intervals:
    plt.axvspan(result_df["timestamp"].iloc[s], result_df["timestamp"].iloc[e], alpha=0.2)

plt.title("Reconstruction error with anomaly intervals")
plt.xlabel("time")
plt.ylabel("reconstruction error")
