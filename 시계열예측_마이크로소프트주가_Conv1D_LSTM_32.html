시계열 데이터셋 만들기¶

In [19]:
def time_dataset(data, timestep, batchsize, shuffle):
    data = tf.expand_dims(data, axis=-1)
    ds = tf.data.Dataset.from_tensor_slices(data)
    ds = ds.window(timestep + 1, shift=1, drop_remainder=True)
    ds = ds.flat_map(lambda w: w.batch(timestep + 1))
    if shuffle:
        ds = ds.shuffle(1000)
    ds = ds.map(lambda w: (w[:-1], w[-1]))
    return ds.batch(batchsize).prefetch(1)
