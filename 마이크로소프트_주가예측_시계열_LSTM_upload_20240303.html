In [169]:
def dataset(data,pred_days,seq_len):
    x = []
    y = []
    for i in range(seq_len, len(data)-pred_days +1):
        x.append(data[i - seq_len:i, 0:data.shape[1]])
        y.append(data[i + pred_days - 1:i + pred_days, 0])  
    return np.array(x), np.array(y)
